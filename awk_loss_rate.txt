BEGIN {

RTO=0.000000
RTT=0.000000
hcount=0
D=0
n=0
       
      }
{
event = $1;
time = $2;
pre = $3;
seq = $4;
nonce = $5;
rtxc = $6;
rto = $7;
hopcount = $8;
sendtime = $9;

 if ( $1 == "Data_Received" ){
 RTO = RTO+$7;
 rtt=$2-$9;
 RTT= RTT+rtt;
 hcount=hcount+$8;
 D= D + (RTT-RTO);
 n=n+1;


    }
      
	  }
	  
END {
printf ("%d %f %f %f %f %f %d\n", $4, hcount/n, RTT/n, RTO/n, D/n, (RTO/n)/(RTT/n), n);
        
}
